openssl genrsa -aes256 -out ca.key 2048
openssl req -new -x509 -days 3650 -key ca.key -sha256 -extensions v3_ca -out ca.crt
openssl req -config site.conf -new -newkey rsa:2048 -nodes -keyout site.key -days 3650 -out site.req
openssl x509 -sha256 -req -in site.req -CA ca.crt -CAkey ca.key -CAcreateserial -out site.crt -days 3650 -extensions req_extensions -extensions cert_extensions -extfile site.conf
